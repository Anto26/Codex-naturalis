package it.polimi.ingsw.gamemodel;

/**

 *
 * @see Board#verifyCardPlacement(it.polimi.ingsw.utils.Pair, Card, Side)
 */
public enum PlacementOutcome {
    VALID,
    INVALID_COORDS,
    INVALID_ENOUGH_RESOURCES
}
